Modified Koch Curve The angles at which the turtle turns are modified, leading the turtle to create a turtle shaped koch curve!

  
  def draw_spesh_koch (t, depth, length):
    if depth > 1:
        t.pencolor("blue")
        draw_koch(t, depth - 1, length)
        t.lt(240)
        draw_koch(t, depth - 1, length)
        t.rt(60)
        draw_koch(t, depth - 1, length)
        t.lt(240)
        draw_koch(t, depth - 1, length)
    else:
        t.pencolor("red")
        t.fd(length)

  def draw_spesh_sierpinski(t, depth, length):
    if depth == 1:
        triangle(t, length)
    else:
        t.pencolor("red")
        draw_sierpinski(t, depth - 1, length / 2)
        t.fd(length / 2)
        t.pencolor("blue")
        draw_sierpinski(t, depth - 1, length / 2)
        t.bk(length / 2)
        t.lt(60)
        t.fd(length / 2)
        t.rt(60)
        t.pencolor("green")
        draw_sierpinski(t, depth - 1, length / 2)
        t.rt(120)
        t.fd(length / 2)
        t.lt(120)

Fractal Tree

Original Fractal Tree tree(t, 4, 50, 30)

Modified Fractal Tree The modified version assigns a random length and random angle of the branches of the tree. 


  def spesh_tree(t, depth, length):
    if depth == 1:
        t.fd(length)
        t.bk(length)
    else:
        random_length = length * random.randint(0, 10)
        random_angle = random.randint (0, 45)
        t.fd(random_length)
        t.rt(random_angle)
        tree(t, depth - 1, random_length, random_angle)
        t.lt(random_angle * 2)
        tree(t, depth - 1, random_length, random_angle)
        t.rt(random_angle)
        t.bk(random_length)
